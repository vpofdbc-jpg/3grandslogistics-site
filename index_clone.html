const express = require('express'); const app = express(); app.use(express.json()); let orders = []; app.post('/create-order', (req, res) => { const { userId, items, address } = req.body; const newOrder = { id: orders.length + 1, userId, items, address, status: 'Pending', createdAt: new Date() }; orders.push(newOrder); res.status(201).json({ message: 'Order created', order: newOrder }); }); app.get('/orders/:id', (req, res) => { const order = orders.find(o => o.id === parseInt(req.params.id)); if (order) { res.json(order); } else { res.status(404).json({ message: 'Order not found' }); } }); app.listen(3000, () => { console.log('Server running on port 3000'); });
